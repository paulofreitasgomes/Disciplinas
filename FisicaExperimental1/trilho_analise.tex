\begin{frame}
\frametitle{Experimento}
\begin{itemize}
\item Uma massa $M$ repousa sobre um trilho de ar sem atrito ligado a outra massa suspensa $m$ suspensa.
\item São medidos os valores de posição e tempo enquanto as massas se deslocam. Com esses, se calculam numericamente também os valores de velocidade e aceleração em função do tempo.
\item A força peso da massa suspensa a puxa para baixo, movendo as duas massas. Aplicando a Lei de Newton em módulo temos:
\begin{equation}
F = mg = (M+m)a \label{acelrelakjhhh}
\end{equation} 
\item Objetivo: calcular a aceleração teórica $a$ usando a Eq. \ref{acelrelakjhhh} e comparar com o valor experimental.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Velocidade e aceleração aproximadas}
\begin{itemize}
\item A definição de velocidade é:
\begin{equation}
v = \dfrac{dx}{dt} = \lim_{\Delta t \rightarrow 0} \dfrac{x(t+\Delta t) - x(t)}{\Delta t} \nonumber
\end{equation} 
\item O que fazemos então é aproximar pelo valor médio no intervalo. Assim, as derivadas para velocidade e aceleração ficam:
\begin{eqnarray}
v(t_i) &\simeq& \bar{v}(t_i) = \dfrac{x(t_{i+1}) - x(t_i) }{\Delta t} \label{velocidadeaproximada} \\
a(t_i) &\simeq& \bar{a}(t_i) = \dfrac{v(t_i) - v(t_{i-1}) }{\Delta t}  \label{aceleracoaproximada}
\end{eqnarray}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Resultados}
\begin{itemize}
\item As massas em gramas são $M=418.3 \pm 0.5$ e $m=10.0 \pm 0.5$. Vamos usar $g=9.80$ m/s$^2$. 
\item Da Eq. \ref{acelrelakjhhh} a aceleração teórica (em m/s$^2$) e o erro propagado na divisão das massa será:
\begin{equation}
a = \dfrac{mg}{M+m} = 0.23 \pm 0.01 
\end{equation}
\item Do gráfico da aceleração, a aceleração experimental é a média $a_m = 0.21 \pm 0.04$ m/s$^2$.
\item O valor experimental pode chegar até 0.25, englobando o valor teórico. 
\item Vemos então que há uma boa concordância entre o experimental e o teórico.
\end{itemize}
\end{frame}



\begin{frame}
\frametitle{Cálculo dos erros}
\begin{eqnarray}
(x \pm \Delta x) + (y \pm \Delta y) &=& (x+y) \pm (\Delta x + \Delta y) \nonumber \\
\dfrac{x \pm \Delta x}{y \pm \Delta y} &=& \dfrac{x}{y} \pm \dfrac{x \Delta y + y \Delta x}{y^2} \nonumber \\
c (x \pm \Delta x)  &=& cx \pm c \Delta x \nonumber \\
\Delta x &=& \sqrt{\dfrac{1}{N-1} \sum_i^N \vert x_i - \bar{x} \vert^2} \nonumber \\
\bar{x} &=& \dfrac{1}{N} \sum_i^N x_i \nonumber 
\end{eqnarray}
\end{frame}


\section{Ajustes}


\begin{frame}
\frametitle{Ajustes linear e não linear}
\begin{itemize} 
\item Movimento de um sistema (composto por dois objetos) submetidos a uma força constante:
\begin{equation}
\vec{F} = m\vec{g} = m \dfrac{d^2 x}{dt^2} \hat{i} = m\vec{a} \label{dkslskwjwiwlkj}
\end{equation}
onde $x(t)$ é a posição do bloco no trilho. Supomos que no instante inicial $t_0=0$ temos $x(0)=x_0 = 0$ e $v(0)=v_0=0$. Resolvendo a equação diferencial \ref{dkslskwjwiwlkj}:
\begin{eqnarray}
x(t) &=& y_0 + v_0 t + \frac{1}{2} a t^2 \label{ajustenaolinear} \\
v(t) &=& v_0 + at \label{ajustelinear} 
\end{eqnarray}
\end{itemize}
\end{frame}

\begin{frame}[fragile] % Need to use the fragile option when verbatim is used in the slide
\frametitle{Coeficientes do ajuste não linear}
\begin{itemize}
\item $\backslash$Z24 define o tamanho da fonte em 24.
\item $\backslash$B define o próximo caracter como sub-índice, como no $f_1$. 
\item $\backslash$S define o próximo caracter como super-índice, como no $t^2$.
\item $\backslash$M volta ao normal e deve ser usado depois do $\backslash$B ou $\backslash$S. Tem que definir o tamanho novamente.
\item Para o símbolo $\pm$, clique em \textit{Special} na esquerda, depois em \textit{Character} e vários símbolos aparecem.
\end{itemize}
\end{frame}
